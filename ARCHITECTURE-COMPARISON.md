        <input

          value="${name}"
          class="profile-input"
          name="full_name"

      </form>
    `;

      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(data)
    });
    
    if (response.ok) {
      this.dispatchEvent(new CustomEvent('profile-updated', { 
        detail: data 
      }));
    }
  }
}
